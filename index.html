    <script>

        let currentLevel = 0;

        let secondsElapsed = 0;

        const correctSound = document.getElementById("sound-correct");

        const wrongSound = document.getElementById("sound-wrong");


        const passwords = ["50", "60", "480", "56", "642", "350", "25", "216", "945.39", "33"];

        const tasks = [

            "<h2>Reproduktionsaufgaben</h2><p>Wie viel sind 25 % von 200?</p>",

            "<p>Ein Hemd kostet 120 €. Der Preis sinkt um 50 %. Wie viel kostet es jetzt?</p>",

            "<p>Ein Fahrrad kostet 400 €. Der Preis steigt um 20 %. Wie hoch ist der neue Preis?</p>",

            "<h3>🔐 Weiter zur direkten Anwendung.</h3><h2>Direkte Anwendung</h2><p>Ein Pullover kostet 80 €. Er ist um 30 % reduziert. Wie viel kostet der Pullover nach dem Rabatt?</p>",

            "<p>Ein Laptop kostet 600 €. Der Preis steigt um 7 %. Wie hoch ist der neue Preis?</p>",

            "<p>Ein Sofa kostet 500 €. Es gibt einen Rabatt von 30 %. Wie viel kostet es jetzt?</p>",

            "<h3>📈 Vertiefung</h3><h2>Prozentuale Veränderung</h2><p>Ein Wert steigt von 120 auf 150. Um wie viel Prozent ist er gestiegen? (ganze Zahl ohne %)</p>",

            "<p>Ein Produkt kostet 200 €. Es wird erst um 10 % reduziert und dann um 20 % erhöht. Was ist der Endpreis?</p>",

            "<p>Ein Smartphone kostet 900 €. Nach einem Rabatt von 10 % wird die Mehrwertsteuer von 19 % hinzugefügt. Wie hoch ist der Endpreis (auf zwei Stellen)?</p>",

            "<h3>🧠 Finale Herausforderung</h3><h2>Erhöhtes Niveau</h2><p>Eine Aktie steigt von 80 € auf 106 €. Um wie viel Prozent ist der Wert gestiegen? (ganze Zahl ohne %)</p>"

        ];


        function checkPassword() {

            let input = document.getElementById("password").value.trim().replace("%", "");

            const feedback = document.getElementById("feedback");


            if (input === passwords[currentLevel]) {

                correctSound.play();

                currentLevel++;

                if (currentLevel < tasks.length) {

                    document.getElementById("task").innerHTML = tasks[currentLevel];

                    document.getElementById("password").value = "";

                    feedback.innerText = "";

                    updateProgress();

                } else {

                    const masterCode = "PZ-8421"; // Dein geheimer Code

document.getElementById("game").innerHTML = `

    <h2>🎉 Glückwunsch! Du hast alle Aufgaben gelöst!</h2>

    <p>⏱️ Deine Zeit: ${secondsElapsed} Sekunden</p>

    <p style="color:#9acd32; font-size: 1.5em; margin-top: 20px;">

        🔑 Dein Schlüsselcode lautet: <strong>${masterCode}</strong>

    </p>

    <p>Trage ihn in dein Lösungsblatt ein oder zeige ihn deiner Lehrkraft.</p>

`;

                }

            } else {

                wrongSound.play();

                feedback.innerText = "❌ Falsche Antwort! Versuche es erneut.";

            }

        }


        function updateProgress() {

            document.getElementById("progress").innerText = `🔐 Aufgabe ${currentLevel + 1} von ${tasks.length}`;

        }


        function startTimer() {

            setInterval(() => {

                secondsElapsed++;

                document.getElementById("timer").innerText = `⏱️ Zeit: ${secondsElapsed} Sekunden`;

            }, 1000);

        }


        window.onload = function () {

            updateProgress();

            startTimer();

            document.getElementById("password").addEventListener("keydown", function (event) {

                if (event.key === "Enter") {

                    checkPassword();

                }

            });

        };
